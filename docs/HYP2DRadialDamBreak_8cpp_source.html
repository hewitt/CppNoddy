38 /// Define the vector flux
39 void flux_fn_x( const DenseVector<double> &x, const DenseVector<double> &q, DenseVector<double> &f ) const
40 {
41 f[ h ] = q[ hu ];
42 f[ hu ] = q[ hu ] * q[ hu ] / q[ h ] + g * q[ h ] * q[ h ] / 2.;
43 f[ hv ] = q[ hu ] * q[ hv ] / q[ h ];
44 }

46 void flux_fn_y( const DenseVector<double> &x, const DenseVector<double> &q, DenseVector<double> &f ) const
47 {
48 f[ h ] = q[ hv ];
49 f[ hu ] = q[ hu ] * q[ hv ] / q[ h ];
50 f[ hv ] = q[ hv ] * q[ hv ] / q[ h ] + g * q[ h ] * q[ h ] / 2.;
51 }

53 // just for fun, we'll use analytic Jacobians
54 void Jac_flux_fn_x( const DenseVector<double> &x, const DenseVector<double> &q, DenseMatrix<double> &J ) const
55 {
56 J( 0, hu ) = 1;
57 J( 1, h ) = - q[ hu ] * q[ hu ] / ( q[ h ] * q[ h ] ) + g * q[ h ];
58 J( 1, hu ) = 2 * q[ hu ] / q[ h ];
59 J( 2, h ) = - q[ hu ] * q[ hv ] / ( q[ h ] * q[ h ] );
60 J( 2, hu ) = q[ hv ] / q[ h ];
61 J( 2, hv ) = q[ hu ] / q[ h ];
62 }
63
64 void Jac_flux_fn_y( const DenseVector<double> &x, const DenseVector<double> &q, DenseMatrix<double> &J ) const
65 {
66 J( 0, hv ) = 1;
67 J( 1, h ) = - q[ hu ] * q[ hv ] / ( q[ h ] * q[ h ] );
68 J( 1, hu ) = q[ hv ] / q[ h ];
69 J( 1, hv ) = q[ hu ] / q[ h ];
70 J( 2, h ) = - q[ hv ] * q[ hv ] / ( q[ h ] * q[ h ] ) + g * q[ h ];
71 J( 2, hv ) = 2 * q[ hv ] / q[ h ];
72 }

74 /// Bound the wave speed
75 void max_charac_speed( const DenseVector<double> &x, const DenseVector<double> &q, DenseVector<double> &c ) const
76 {
77 // wave speed
